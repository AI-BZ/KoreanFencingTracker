# AI 자동 주문 플로우
class AutoOrderSystem:
    def __init__(self):
        self.suppliers = {
            'primary': NeedABladeAPI(),
            'backup1': FencingPostAPI(),
            'backup2': AmazonAPI(),
            'emergency': PollandScraper()
        }
        self.forwarders = {
            'us': MyUSAddress(),
            'pl': Polbox()
        }
    
    async def process_order(self, order):
        # 1. 재고 확인 (병렬)
        availability = await self.check_all_suppliers(order.sku)
        
        # 2. 최적 공급처 선택
        supplier = self.select_best_supplier(availability, {
            'priority': ['price', 'speed', 'reliability'],
            'weights': [0.3, 0.3, 0.4]
        })
        
        # 3. 자동 주문
        order_result = await supplier.place_order(order)
        
        # 4. 배송대행 자동 처리
        if supplier.country == 'US':
            await self.forwarders['us'].create_shipment(order_result)
        
        # 5. 고객 알림
        await self.notify_customer(order, estimated_delivery)
        
        return order_result

    def select_best_supplier(self, availability, criteria):
        scores = {}
        for supplier, data in availability.items():
            if not data['in_stock']:
                continue
            
            scores[supplier] = {
                'price_score': (150 - data['price']) / 50,
                'speed_score': (10 - data['delivery_days']) / 5,
                'reliability_score': data['success_rate']
            }
        
        # 가중치 적용
        best = max(scores.items(), 
                  key=lambda x: sum(x[1][k] * criteria['weights'][i] 
                                  for i, k in enumerate(x[1].keys())))
        return self.suppliers[best[0]]

class KFAFencingScraper:
    """대한펜싱협회 데이터 수집"""
    
    def __init__(self):
        self.base_url = "https://fencing.sports.or.kr"
        self.headers = {
            'User-Agent': 'Mozilla/5.0 (교육/연구 목적)',
            'Referer': 'https://fencing.sports.or.kr'
        }
        self.delay = 2  # 서버 부하 방지
    
    def get_competition_list(self):
        """대회 목록 가져오기"""
        url = f"{self.base_url}/game/compListView?code=game"
        
        try:
            # 정중한 요청
            time.sleep(self.delay)
            response = requests.get(url, headers=self.headers)
            
            if response.status_code == 200:
                soup = BeautifulSoup(response.text, 'html.parser')
                
                # 테이블 파싱 예시
                competitions = []
                table = soup.find('table', class_='competition-table')
                
                if table:
                    rows = table.find_all('tr')[1:]  # 헤더 제외
                    for row in rows:
                        cols = row.find_all('td')
                        if cols:
                            competition = {
                                'name': cols[0].text.strip(),
                                'date': cols[1].text.strip(),
                                'location': cols[2].text.strip(),
                                'source': '대한펜싱협회'
                            }
                            competitions.append(competition)
                
                return competitions
        
        except Exception as e:
            print(f"Error: {e}")
            return []

class DynamicScraper:
    """동적 페이지 스크래핑"""
    
    def __init__(self):
        options = webdriver.ChromeOptions()
        options.add_argument('--headless')  # 백그라운드 실행
        self.driver = webdriver.Chrome(options=options)
    
    def get_ajax_data(self):
        """AJAX 로드 데이터 가져오기"""
        self.driver.get("https://fencing.sports.or.kr/game/compListView")
        
        # AJAX 로드 대기
        wait = WebDriverWait(self.driver, 10)
        table = wait.until(
            EC.presence_of_element_located((By.CLASS_NAME, "result-table"))
        )
        
        # 네트워크 로그 확인 (API 찾기)
        logs = self.driver.get_log('performance')
        for log in logs:
            message = json.loads(log['message'])
            if 'Network.response' in message['message']['method']:
                url = message['message']['params'].get('response', {}).get('url', '')
                if 'api' in url or 'json' in url:
                    print(f"Found API: {url}")
        
        return table.text

// Console에서 직접 API 호출 테스트

// 1. GET 요청 테스트
async function testAPI() {
    const possibleURLs = [
        '/game/selectGameList.do',
        '/game/getCompList.json',
        '/api/competitions',
        '/data/game/list'
    ];
    
    for(let url of possibleURLs) {
        try {
            const response = await fetch(url, {
                method: 'GET',
                headers: {
                    'X-Requested-With': 'XMLHttpRequest',
                    'Accept': 'application/json'
                }
            });
            
            if(response.ok) {
                console.log(`✅ Found API: ${url}`);
                const data = await response.json();
                console.log(data);
                return data;
            }
        } catch(e) {
            console.log(`❌ Failed: ${url}`);
        }
    }
}

// 2. POST 요청 테스트 (리스트 조회)
async function testPostAPI() {
    const url = '/game/selectGameList.do';
    const params = {
        pageIndex: 1,
        pageSize: 10,
        searchCondition: '',
        searchKeyword: ''
    };
    
    try {
        const response = await fetch(url, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded',
                'X-Requested-With': 'XMLHttpRequest'
            },
            body: new URLSearchParams(params)
        });
        
        const data = await response.json();
        console.log('API Response:', data);
        return data;
    } catch(e) {
        console.error('API Error:', e);
    }
}
